df = df.dropna(subset=['question', 'label_str'])

labels = df['label_str'].unique().tolist()
label2id = {label: i for i, label in enumerate(labels)}
id2label = {i: label for i, label in enumerate(labels)}

def extract_clean_code(content):
    tick = chr(96)
    marker = tick * 3
    pattern = rf"{marker}(?:python)?\s*(.*?){marker}"
    blocks = re.findall(pattern, content, re.DOTALL | re.IGNORECASE)

    if blocks:
        for block in reversed(blocks):
            if 'final_result' in block:
                return block.strip()
        return blocks[-1].strip()

    lines = content.split('\n')
    code_lines = []
    for line in lines:
        if not line.startswith('#') and not line.startswith(' ') and '=' not in line and 'import' not in line:
            if not any(char.isdigit() for char in line):
                continue
        code_lines.append(line)

    clean_code = "\n".join(code_lines).strip()
    return clean_code.replace(f"{marker}python", "").replace(marker, "")

def extract_clean_code(content):
    tick = chr(96)
    marker = tick * 3
    pattern = rf"{marker}(?:python)?\s*(.*?){marker}"
    blocks = re.findall(pattern, content, re.DOTALL | re.IGNORECASE)

    if blocks:
        for block in reversed(blocks):
            if 'final_result' in block:
                return block.strip()
        return blocks[-1].strip()

    lines = content.split('\n')
    code_lines = []
    for line in lines:
        if not line.startswith('#') and not line.startswith(' ') and '=' not in line and 'import' not in line:
            if not any(char.isdigit() for char in line):
                continue
        code_lines.append(line)

    clean_code = "\n".join(code_lines).strip()
    return clean_code.replace(f"{marker}python", "").replace(marker, "")

available_df = available_df.dropna(subset=['question', 'answer'])

# Lấy 7 dạng bài hợp lệ (Bỏ CHLT)
all_golden_prefixes = golden_df['prefix'].unique().tolist()
valid_prefixes = [p for p in all_golden_prefixes if p != 'CHLT']

def get_prefix(id_val):
    return "".join([c for c in str(id_val) if c.isalpha()])

available_df['prefix'] = available_df['id'].apply(get_prefix)
available_df = available_df[available_df['prefix'].isin(valid_prefixes)]

# Dynamic K-means
SAMPLING_RATIO = 0.20  # 20% for each
MIN_CLUSTERS = 5       # min = 5 questions
MAX_CLUSTERS = 30      # max = 30 questions

selected_candidates = []

print("\n🧠 Đang phân tích quy mô và gom cụm...")

for prefix in valid_prefixes:
    subset = available_df[available_df['prefix'] == prefix].copy()
    n_samples = len(subset)

    if n_samples == 0:
        continue

    # Calculate k for each
    k_dynamic = int(n_samples * SAMPLING_RATIO)
    k_dynamic = max(MIN_CLUSTERS, min(k_dynamic, MAX_CLUSTERS)) # Min-Max boundary

    if n_samples <= k_dynamic:
        selected_candidates.append(subset)
        print(f"✔️ {prefix} (Tổng: {n_samples}): Lấy toàn bộ (Nhỏ hơn mức quy định {k_dynamic}).")
        continue

    print(f"⏳ {prefix} (Tổng: {n_samples}): Trích xuất {SAMPLING_RATIO*100}% -> Gom thành {k_dynamic} cụm tinh hoa...")

    texts = subset['question'].tolist()
    embeddings = model.encode(texts, show_progress_bar=False)

    kmeans = KMeans(n_clusters=k_dynamic, random_state=42, n_init=10)
    kmeans.fit(embeddings)

    closest_indices, _ = pairwise_distances_argmin_min(kmeans.cluster_centers_, embeddings)

    bright_candidates = subset.iloc[closest_indices]
    selected_candidates.append(bright_candidates)

# Export
final_candidates_df = pd.concat(selected_candidates, ignore_index=True)
final_candidates_df = final_candidates_df.sample(frac=1, random_state=42).reset_index(drop=True)
